# 4. GEOMETRY HELPERS

def line_parts(geom):
    """Return list of coordinate sequences for LineString or MultiLineString."""
    if geom is None or geom.is_empty:
        return []
    gtype = geom.geom_type
    if gtype == "LineString":
        x, y = geom.xy
        return [list(zip(x, y))]
    elif gtype == "MultiLineString":
        parts = []
        for part in geom.geoms:
            x, y = part.xy
            parts.append(list(zip(x, y)))
        return parts
    return []

def polygon_rings(geom):
    """Return exterior rings for Polygon or MultiPolygon."""
    if geom is None or geom.is_empty:
        return []
    gtype = geom.geom_type
    rings = []
    if gtype == "Polygon":
        x, y = geom.exterior.xy
        rings.append(list(zip(x, y)))
    elif gtype == "MultiPolygon":
        for poly in geom.geoms:
            x, y = poly.exterior.xy
            rings.append(list(zip(x, y)))
    return rings

def rings_to_lonlat_lists(rings):
    """Convert multiple rings into lon/lat lists separated by None."""
    lons, lats = [], []
    for ring in rings:
        for lon, lat in ring:
            lons.append(lon)
            lats.append(lat)
        lons.append(None)
        lats.append(None)
    return lons, lats

    # turn on selected mode

        vis[idx] = True

    return vis
